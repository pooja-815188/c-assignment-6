#include <stdio.h>
#include <string.h>
 
int main()
{
  	char str[100], ch, Newch;
  	int i;
 
  	printf(" Enter any String\n");
  	gets(str);
  	printf("Enter the Character to Search for\n");
  	scanf("%c", &ch);
  	getchar();
  	printf("\n Enter the New Character :  ");
  	scanf("%c", &Newch);
  	
  	for(i = 0; i <= strlen(str); i++)
  	{
  		if(str[i] == ch)  
		{
  			str[i] = Newch;
 		}
	}
	
	printf("\n The Final String after Replacing '%c' with '%c' = %s ", ch, Newch, str);
}
